36template<class ActionT>
37BtActionServer<ActionT>::BtActionServer(
38 const rclcpp_lifecycle::LifecycleNode::WeakPtr & parent,
39 const std::string & action_name,
40 const std::vector<std::string> & plugin_lib_names,
41 const std::string & default_bt_xml_filename,
42 OnGoalReceivedCallback on_goal_received_callback,
43 OnLoopCallback on_loop_callback,
44 OnPreemptCallback on_preempt_callback,
45 OnCompletionCallback on_completion_callback)
46: action_name_(action_name),
47 default_bt_xml_filename_(default_bt_xml_filename),
48 plugin_lib_names_(plugin_lib_names),
49 node_(parent),
50 on_goal_received_callback_(on_goal_received_callback),
51 on_loop_callback_(on_loop_callback),
52 on_preempt_callback_(on_preempt_callback),
53 on_completion_callback_(on_completion_callback)
54{
55 auto node = node_.lock();
56 logger_ = node->get_logger();
57 clock_ = node->get_clock();
58
59 // Declare this node's parameters
60 if (!node->has_parameter("bt_loop_duration")) {
61 node->declare_parameter("bt_loop_duration", 10);
62 }
63 if (!node->has_parameter("default_server_timeout")) {
64 node->declare_parameter("default_server_timeout", 20);
65 }
66 if (!node->has_parameter("action_server_result_timeout")) {
67 node->declare_parameter("action_server_result_timeout", 900.0);
68 }
69 if (!node->has_parameter("always_reload_bt_xml")) {
70 node->declare_parameter("always_reload_bt_xml", false);
71 }
72 if (!node->has_parameter("wait_for_service_timeout")) {
73 node->declare_parameter("wait_for_service_timeout", 1000);
74 }
75
76 std::vector<std::string> error_code_names = {
77 "follow_path_error_code",
78 "compute_path_error_code"
79 };
80
81 if (!node->has_parameter("error_code_names")) {
82 const rclcpp::ParameterValue value = node->declare_parameter(
83 "error_code_names",
84 rclcpp::PARAMETER_STRING_ARRAY);
85 if (value.get_type() == rclcpp::PARAMETER_NOT_SET) {
86 std::string error_codes_str;
87 for (const auto & error_code : error_code_names) {
88 error_codes_str += " " + error_code;
89 }
90 RCLCPP_WARN_STREAM(
91 logger_, "Error_code parameters were not set. Using default values of:"
92 << error_codes_str + "\n"
93 << "Make sure these match your BT and there are not other sources of error codes you"
94 "reported to your application");
95 rclcpp::Parameter error_code_names_param("error_code_names", error_code_names);
96 node->set_parameter(error_code_names_param);
97 } else {
98 error_code_names = value.get<std::vector<std::string>>();
99 std::string error_codes_str;
100 for (const auto & error_code : error_code_names) {
101 error_codes_str += " " + error_code;
102 }
103 RCLCPP_INFO_STREAM(logger_, "Error_code parameters were set to:" << error_codes_str);
104 }
105 }
106}

112template<class ActionT>
113bool BtActionServer<ActionT>::on_configure()
114{
115 auto node = node_.lock();
116 if (!node) {
117 throw std::runtime_error{"Failed to lock node"};
118 }
119
120 // Name client node after action name
121 std::string client_node_name = action_name_;
122 std::replace(client_node_name.begin(), client_node_name.end(), '/', '_');
123 // Use suffix '_rclcpp_node' to keep parameter file consistency #1773
124 auto options = rclcpp::NodeOptions().arguments(
125 {"--ros-args",
126 "-r",
127 std::string("__node:=") +
128 std::string(node->get_name()) + "_" + client_node_name + "_rclcpp_node",
129 "-p",
130 "use_sim_time:=" +
131 std::string(node->get_parameter("use_sim_time").as_bool() ? "true" : "false"),
132 "--"});
133
134 // Support for handling the topic-based goal pose from rviz
135 client_node_ = std::make_shared<rclcpp::Node>("_", options);
136
137 // Declare parameters for common client node applications to share with BT nodes
138 // Declare if not declared in case being used an external application, then copying
139 // all of the main node's parameters to the client for BT nodes to obtain
140 nav2_util::declare_parameter_if_not_declared(
141 node, "global_frame", rclcpp::ParameterValue(std::string("map")));
142 nav2_util::declare_parameter_if_not_declared(
143 node, "robot_base_frame", rclcpp::ParameterValue(std::string("base_link")));
144 nav2_util::declare_parameter_if_not_declared(
145 node, "transform_tolerance", rclcpp::ParameterValue(0.1));
146 rclcpp::copy_all_parameter_values(node, client_node_);
147
148 // set the timeout in seconds for the action server to discard goal handles if not finished
149 double action_server_result_timeout =
150 node->get_parameter("action_server_result_timeout").as_double();
151 rcl_action_server_options_t server_options = rcl_action_server_get_default_options();
152 server_options.result_timeout.nanoseconds = RCL_S_TO_NS(action_server_result_timeout);
153
154 action_server_ = std::make_shared<ActionServer>(
155 node->get_node_base_interface(),
156 node->get_node_clock_interface(),
157 node->get_node_logging_interface(),
158 node->get_node_waitables_interface(),
159 action_name_, std::bind(&BtActionServer<ActionT>::executeCallback, this),
160 nullptr, std::chrono::milliseconds(500), false, server_options);
161
162 // Get parameters for BT timeouts
163 int bt_loop_duration;
164 node->get_parameter("bt_loop_duration", bt_loop_duration);
165 bt_loop_duration_ = std::chrono::milliseconds(bt_loop_duration);
166 int default_server_timeout;
167 node->get_parameter("default_server_timeout", default_server_timeout);
168 default_server_timeout_ = std::chrono::milliseconds(default_server_timeout);
169 int wait_for_service_timeout;
170 node->get_parameter("wait_for_service_timeout", wait_for_service_timeout);
171 wait_for_service_timeout_ = std::chrono::milliseconds(wait_for_service_timeout);
172 node->get_parameter("always_reload_bt_xml", always_reload_bt_xml_);
173
174 // Get error code id names to grab off of the blackboard
175 error_code_names_ = node->get_parameter("error_code_names").as_string_array();
176
177 // Create the class that registers our custom nodes and executes the BT
178 bt_ = std::make_unique<nav2_behavior_tree::BehaviorTreeEngine>(plugin_lib_names_, client_node_);
179
180 // Create the blackboard that will be shared by all of the nodes in the tree
181 blackboard_ = BT::Blackboard::create();
182
183 // Put items on the blackboard
184 blackboard_->set<rclcpp::Node::SharedPtr>("node", client_node_); // NOLINT
185 blackboard_->set<std::chrono::milliseconds>("server_timeout", default_server_timeout_); // NOLINT
186 blackboard_->set<std::chrono::milliseconds>("bt_loop_duration", bt_loop_duration_); // NOLINT
187 blackboard_->set<std::chrono::milliseconds>(
188 "wait_for_service_timeout",
189 wait_for_service_timeout_);
190
191 return true;
192}

226template<class ActionT>
227bool BtActionServer<ActionT>::loadBehaviorTree(const std::string & bt_xml_filename)
228{
229 // Empty filename is default for backward compatibility
230 auto filename = bt_xml_filename.empty() ? default_bt_xml_filename_ : bt_xml_filename;
231
232 // Use previous BT if it is the existing one and always reload flag is not set to true
233 if (!always_reload_bt_xml_ && current_bt_xml_filename_ == filename) {
234 RCLCPP_DEBUG(logger_, "BT will not be reloaded as the given xml is already loaded");
235 return true;
236 }
237
238 // Read the input BT XML from the specified file into a string
239 std::ifstream xml_file(filename);
240
241 if (!xml_file.good()) {
242 RCLCPP_ERROR(logger_, "Couldn't open input XML file: %s", filename.c_str());
243 return false;
244 }
245
246 // Create the Behavior Tree from the XML input
247 try {
248 tree_ = bt_->createTreeFromFile(filename, blackboard_);
249 for (auto & subtree : tree_.subtrees) {
250 auto & blackboard = subtree->blackboard;
251 blackboard->set("node", client_node_);
252 blackboard->set<std::chrono::milliseconds>("server_timeout", default_server_timeout_);
253 blackboard->set<std::chrono::milliseconds>("bt_loop_duration", bt_loop_duration_);
254 blackboard->set<std::chrono::milliseconds>(
255 "wait_for_service_timeout",
256 wait_for_service_timeout_);
257 }
258 } catch (const std::exception & e) {
259 RCLCPP_ERROR(logger_, "Exception when loading BT: %s", e.what());
260 return false;
261 }
262
263 topic_logger_ = std::make_unique<RosTopicLogger>(client_node_, tree_);
264
265 current_bt_xml_filename_ = filename;
266 return true;
267}

269template<class ActionT>
270void BtActionServer<ActionT>::executeCallback()
271{
272 if (!on_goal_received_callback_(action_server_->get_current_goal())) {
273 action_server_->terminate_current();
274 cleanErrorCodes();
275 return;
276 }
277
278 auto is_canceling = [&]() {
279 if (action_server_ == nullptr) {
280 RCLCPP_DEBUG(logger_, "Action server unavailable. Canceling.");
281 return true;
282 }
283 if (!action_server_->is_server_active()) {
284 RCLCPP_DEBUG(logger_, "Action server is inactive. Canceling.");
285 return true;
286 }
287 return action_server_->is_cancel_requested();
288 };
289
290 auto on_loop = [&]() {
291 if (action_server_->is_preempt_requested() && on_preempt_callback_) {
292 on_preempt_callback_(action_server_->get_pending_goal());
293 }
294 topic_logger_->flush();
295 on_loop_callback_();
296 };
297
298 // Execute the BT that was previously created in the configure step
299 nav2_behavior_tree::BtStatus rc = bt_->run(&tree_, on_loop, is_canceling, bt_loop_duration_);
300
301 // Make sure that the Bt is not in a running state from a previous execution
302 // note: if all the ControlNodes are implemented correctly, this is not needed.
303 bt_->haltAllActions(tree_);
304
305 // Give server an opportunity to populate the result message or simple give
306 // an indication that the action is complete.
307 auto result = std::make_shared<typename ActionT::Result>();
308
309 populateErrorCode(result);
310
311 on_completion_callback_(result, rc);
312
313 switch (rc) {
314 case nav2_behavior_tree::BtStatus::SUCCEEDED:
315 action_server_->succeeded_current(result);
316 RCLCPP_INFO(logger_, "Goal succeeded");
317 break;
318
319 case nav2_behavior_tree::BtStatus::FAILED:
320 action_server_->terminate_current(result);
321 RCLCPP_ERROR(logger_, "Goal failed");
322 break;
323
324 case nav2_behavior_tree::BtStatus::CANCELED:
325 action_server_->terminate_all(result);
326 RCLCPP_INFO(logger_, "Goal canceled");
327 break;
328 }
329
330 cleanErrorCodes();
331}

333template<class ActionT>
334void BtActionServer<ActionT>::populateErrorCode(
335 typename std::shared_ptr<typename ActionT::Result> result)
336{
337 int highest_priority_error_code = std::numeric_limits<int>::max();
338 for (const auto & error_code : error_code_names_) {
339 try {
340 int current_error_code = blackboard_->get<int>(error_code);
341 if (current_error_code != 0 && current_error_code < highest_priority_error_code) {
342 highest_priority_error_code = current_error_code;
343 }
344 } catch (...) {
345 RCLCPP_DEBUG(
346 logger_,
347 "Failed to get error code: %s from blackboard",
348 error_code.c_str());
349 }
350 }
351
352 if (highest_priority_error_code != std::numeric_limits<int>::max()) {
353 result->error_code = highest_priority_error_code;
354 }
355}
