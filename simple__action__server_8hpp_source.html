41 // Callback function to complete main work. This should itself deal with its
42 // own exceptions, but if for some reason one is thrown, it will be caught
43 // in SimpleActionServer and terminate the action itself.
44 typedef std::function<void ()> ExecuteCallback;
45
46 // Callback function to notify the user that an exception was thrown that
47 // the simple action server caught (or another failure) and the action was
48 // terminated. To avoid using, catch exceptions in your application such that
49 // the SimpleActionServer will never need to terminate based on failed action
50 // ExecuteCallback.
51 typedef std::function<void ()> CompletionCallback;

197 void handle_accepted(const std::shared_ptr<rclcpp_action::ServerGoalHandle<ActionT>> handle)
198 {
199 std::lock_guard<std::recursive_mutex> lock(update_mutex_);
200 debug_msg("Receiving a new goal");
201
202 if (is_active(current_handle_) || is_running()) {
203 debug_msg("An older goal is active, moving the new goal to a pending slot.");
204
205 if (is_active(pending_handle_)) {
206 debug_msg(
207 "The pending slot is occupied."
208 " The previous pending goal will be terminated and replaced.");
209 terminate(pending_handle_);
210 }
211 pending_handle_ = handle;
212 preempt_requested_ = true;
213 } else {
214 if (is_active(pending_handle_)) {
215 // Shouldn't reach a state with a pending goal but no current one.
216 error_msg("Forgot to handle a preemption. Terminating the pending goal.");
217 terminate(pending_handle_);
218 preempt_requested_ = false;
219 }
220
221 current_handle_ = handle;
222
223 // Return quickly to avoid blocking the executor, so spin up a new thread
224 debug_msg("Executing goal asynchronously.");
225 execution_future_ = std::async(
226 std::launch::async, [this]() {
227 setSoftRealTimePriority();
228 work();
229 });
230 }
231 }
232
236 void work()
237 {
238 while (rclcpp::ok() && !stop_execution_ && is_active(current_handle_)) {
239 debug_msg("Executing the goal...");
240 try {
241 execute_callback_();
242 } catch (std::exception & ex) {
243 RCLCPP_ERROR(
244 node_logging_interface_->get_logger(),
245 "Action server failed while executing action callback: \"%s\"", ex.what());
246 terminate_all();
247 if (completion_callback_) {completion_callback_();}
248 return;
249 }
250
251 debug_msg("Blocking processing of new goal handles.");
252 std::lock_guard<std::recursive_mutex> lock(update_mutex_);
253
254 if (stop_execution_) {
255 warn_msg("Stopping the thread per request.");
256 terminate_all();
257 if (completion_callback_) {completion_callback_();}
258 break;
259 }
260
261 if (is_active(current_handle_)) {
262 warn_msg("Current goal was not completed successfully.");
263 terminate(current_handle_);
264 if (completion_callback_) {completion_callback_();}
265 }
266
267 if (is_active(pending_handle_)) {
268 debug_msg("Executing a pending handle on the existing thread.");
269 accept_pending_goal();
270 } else {
271 debug_msg("Done processing available goals.");
272 break;
273 }
274 }
275 debug_msg("Worker thread done.");
276 }

291 void deactivate()
292 {
293 debug_msg("Deactivating...");
294
295 {
296 std::lock_guard<std::recursive_mutex> lock(update_mutex_);
297 server_active_ = false;
298 stop_execution_ = true;
299 }
300
301 if (!execution_future_.valid()) {
302 return;
303 }
304
305 if (is_running()) {
306 warn_msg(
307 "Requested to deactivate server but goal is still executing."
308 " Should check if action server is running before deactivating.");
309 }
310
311 using namespace std::chrono; //NOLINT
312 auto start_time = steady_clock::now();
313 while (execution_future_.wait_for(milliseconds(100)) != std::future_status::ready) {
314 info_msg("Waiting for async process to finish.");
315 if (steady_clock::now() - start_time >= server_timeout_) {
316 terminate_all();
317 if (completion_callback_) {completion_callback_();}
318 error_msg("Action callback is still running and missed deadline to stop");
319 }
320 }
321
322 debug_msg("Deactivation completed.");
323 }

329 bool is_running()
330 {
331 return execution_future_.valid() &&
332 (execution_future_.wait_for(std::chrono::milliseconds(0)) ==
333 std::future_status::timeout);
334 }

360 const std::shared_ptr<const typename ActionT::Goal> accept_pending_goal()
361 {
362 std::lock_guard<std::recursive_mutex> lock(update_mutex_);
363
364 if (!pending_handle_ || !pending_handle_->is_active()) {
365 error_msg("Attempting to get pending goal when not available");
366 return std::shared_ptr<const typename ActionT::Goal>();
367 }
368
369 if (is_active(current_handle_) && current_handle_ != pending_handle_) {
370 debug_msg("Cancelling the previous goal");
371 current_handle_->abort(empty_result());
372 }
373
374 current_handle_ = pending_handle_;
375 pending_handle_.reset();
376 preempt_requested_ = false;
377
378 debug_msg("Preempted goal");
379
380 return current_handle_->get_goal();
381 }

386 void terminate_pending_goal()

389
390 if (!pending_handle_ || !pending_handle_->is_active()) {
391 error_msg("Attempting to terminate pending goal when not available");
392 return;
393 }

395 terminate(pending_handle_);
396 preempt_requested_ = false;
397
398 debug_msg("Pending goal terminated");
399 }

449 bool is_cancel_requested() const
450 {
451 std::lock_guard<std::recursive_mutex> lock(update_mutex_);
452
453 // A cancel request is assumed if either handle is canceled by the client.
454
455 if (current_handle_ == nullptr) {
456 error_msg("Checking for cancel but current goal is not available");
457 return false;
458 }
459
460 if (pending_handle_ != nullptr) {
461 return pending_handle_->is_canceling();
462 }
463
464 return current_handle_->is_canceling();

576 void terminate(
577 std::shared_ptr<rclcpp_action::ServerGoalHandle<ActionT>> & handle,
578 typename std::shared_ptr<typename ActionT::Result> result =
579 std::make_shared<typename ActionT::Result>())
580 {
581 std::lock_guard<std::recursive_mutex> lock(update_mutex_);
582
583 if (is_active(handle)) {
584 if (handle->is_canceling()) {
585 warn_msg("Client requested to cancel the goal. Cancelling.");
586 handle->canceled(result);
587 } else {
588 warn_msg("Aborting handle.");
589 handle->abort(result);
590 }
591 handle.reset();
592 }
