100 bool Evaluate(
101 double const * const * parameters,
102 double * residuals,
103 double ** jacobians) const override
104 {
105 Eigen::Vector2d xi = Eigen::Vector2d(parameters[0][0], parameters[1][0]);
106 Eigen::Vector2d xi_p1 = _path.at(index + 1);
107 Eigen::Vector2d xi_m1 = _path.at(index - 1);
108 CurvatureComputations curvature_params;
109 double grad_x_raw = 0, grad_y_raw = 0, cost_raw = 0;
110
111 // compute cost
112 addSmoothingResidual(15000, xi, xi_p1, xi_m1, cost_raw);
113 addCurvatureResidual(60.0, xi, xi_p1, xi_m1, curvature_params, cost_raw);
114
115 residuals[0] = 0;
116 residuals[0] = cost_raw; // objective function value x
117
118 if (jacobians != NULL && jacobians[0] != NULL) {
119 addSmoothingJacobian(15000, xi, xi_p1, xi_m1, grad_x_raw, grad_y_raw);
120 addCurvatureJacobian(60.0, xi, xi_p1, xi_m1, curvature_params, grad_x_raw, grad_y_raw);
121
122 jacobians[0][0] = 0;
123 jacobians[1][0] = 0;
124 jacobians[0][0] = grad_x_raw; // x derivative
125 jacobians[1][0] = grad_y_raw; // y derivative
126 jacobians[0][1] = 0.0;
127 jacobians[1][1] = 0.0;
128 }
129
130 return true;
131 }

142 inline void addSmoothingResidual(
143 const double & weight,
144 const Eigen::Vector2d & pt,
145 const Eigen::Vector2d & pt_p,
146 const Eigen::Vector2d & pt_m,
147 double & r) const
148 {
149 r += weight * (
150 pt_p.dot(pt_p) -
151 4 * pt_p.dot(pt) +
152 2 * pt_p.dot(pt_m) +
153 4 * pt.dot(pt) -
154 4 * pt.dot(pt_m) +
155 pt_m.dot(pt_m)); // objective function value
156 }

167 inline void addSmoothingJacobian(
168 const double & weight,
169 const Eigen::Vector2d & pt,
170 const Eigen::Vector2d & pt_p,
171 const Eigen::Vector2d & pt_m,
172 double & j0,
173 double & j1) const
174 {
175 j0 += weight *
176 (-4 * pt_m[0] + 8 * pt[0] - 4 * pt_p[0]); // xi x component of partial-derivative
177 j1 += weight *
178 (-4 * pt_m[1] + 8 * pt[1] - 4 * pt_p[1]); // xi y component of partial-derivative
179 }

188 inline void getCurvatureParams(
189 const Eigen::Vector2d & pt,
190 const Eigen::Vector2d & pt_p,
191 const Eigen::Vector2d & pt_m,
192 CurvatureComputations & curvature_params) const
193 {
194 curvature_params.valid = true;
195 curvature_params.delta_xi = Eigen::Vector2d(pt[0] - pt_m[0], pt[1] - pt_m[1]);
196 curvature_params.delta_xi_p = Eigen::Vector2d(pt_p[0] - pt[0], pt_p[1] - pt[1]);
197 curvature_params.delta_xi_norm = curvature_params.delta_xi.norm();
198 curvature_params.delta_xi_p_norm = curvature_params.delta_xi_p.norm();
199
200 if (curvature_params.delta_xi_norm < EPSILON || curvature_params.delta_xi_p_norm < EPSILON ||
201 std::isnan(curvature_params.delta_xi_p_norm) || std::isnan(curvature_params.delta_xi_norm) ||
202 std::isinf(curvature_params.delta_xi_p_norm) || std::isinf(curvature_params.delta_xi_norm))
203 {
204 // ensure we have non-nan values returned
205 curvature_params.valid = false;
206 return;
207 }
208
209 const double & delta_xi_by_xi_p =
210 curvature_params.delta_xi_norm * curvature_params.delta_xi_p_norm;
211 double projection =
212 curvature_params.delta_xi.dot(curvature_params.delta_xi_p) / delta_xi_by_xi_p;
213 if (fabs(1 - projection) < EPSILON || fabs(projection + 1) < EPSILON) {
214 projection = 1.0;
215 }
216
217 curvature_params.delta_phi_i = std::acos(projection);
218 curvature_params.turning_rad = curvature_params.delta_phi_i / curvature_params.delta_xi_norm;
219
220 curvature_params.ki_minus_kmax = curvature_params.turning_rad - _upsample_ratio *
221 _params.max_curvature;
222
223 if (curvature_params.ki_minus_kmax <= EPSILON) {
224 // Quadratic penalty need not apply
225 curvature_params.valid = false;
226 }
227 }

238 inline void addCurvatureResidual(
239 const double & weight,
240 const Eigen::Vector2d & pt,
241 const Eigen::Vector2d & pt_p,
242 const Eigen::Vector2d & pt_m,
243 CurvatureComputations & curvature_params,
244 double & r) const
245 {
246 getCurvatureParams(pt, pt_p, pt_m, curvature_params);
247
248 if (!curvature_params.isValid()) {
249 return;
250 }
251
252 // objective function value
253 r += weight *
254 curvature_params.ki_minus_kmax * curvature_params.ki_minus_kmax;
255 }

267 inline void addCurvatureJacobian(
268 const double & weight,
269 const Eigen::Vector2d & pt,
270 const Eigen::Vector2d & pt_p,
271 const Eigen::Vector2d & /*pt_m*/,
272 CurvatureComputations & curvature_params,
273 double & j0,
274 double & j1) const
275 {
276 if (!curvature_params.isValid()) {
277 return;
278 }
279
280 const double & partial_delta_phi_i_wrt_cost_delta_phi_i =
281 -1 / std::sqrt(1 - std::pow(std::cos(curvature_params.delta_phi_i), 2));
282 // const Eigen::Vector2d ones = Eigen::Vector2d(1.0, 1.0);
283 auto neg_pt_plus = -1 * pt_p;
284 Eigen::Vector2d p1 = normalizedOrthogonalComplement(
285 pt, neg_pt_plus, curvature_params.delta_xi_norm, curvature_params.delta_xi_p_norm);
286 Eigen::Vector2d p2 = normalizedOrthogonalComplement(
287 neg_pt_plus, pt, curvature_params.delta_xi_p_norm, curvature_params.delta_xi_norm);
288
289 const double & u = 2 * curvature_params.ki_minus_kmax;
290 const double & common_prefix =
291 (1 / curvature_params.delta_xi_norm) * partial_delta_phi_i_wrt_cost_delta_phi_i;
292 const double & common_suffix = curvature_params.delta_phi_i /
293 (curvature_params.delta_xi_norm * curvature_params.delta_xi_norm);
294 const Eigen::Vector2d & d_delta_xi_d_xi = curvature_params.delta_xi /
295 curvature_params.delta_xi_norm;
296
297 const Eigen::Vector2d jacobian = u *
298 (common_prefix * (-p1 - p2) - (common_suffix * d_delta_xi_d_xi));
299 const Eigen::Vector2d jacobian_im1 = u *
300 (common_prefix * p2 + (common_suffix * d_delta_xi_d_xi));
301 const Eigen::Vector2d jacobian_ip1 = u * (common_prefix * p1);
302 j0 += weight * jacobian[0]; // xi x component of partial-derivative
303 j1 += weight * jacobian[1]; // xi y component of partial-derivative
304 // j0 += weight *
305 // (jacobian_im1[0] + 2 * jacobian[0] + jacobian_ip1[0]);
306 // j1 += weight *
307 // (jacobian_im1[1] + 2 * jacobian[1] + jacobian_ip1[1]);
308 }
317 inline Eigen::Vector2d normalizedOrthogonalComplement(
318 const Eigen::Vector2d & a,
319 const Eigen::Vector2d & b,
320 const double & a_norm,
321 const double & b_norm) const
322 {
323 return (a - (a.dot(b) * b / b.squaredNorm())) / (a_norm * b_norm);
324 }
