33char printableCost(unsigned char cost)
34{
35 switch (cost) {
36 case nav2_costmap_2d::NO_INFORMATION: return '?';
37 case nav2_costmap_2d::LETHAL_OBSTACLE: return 'L';
38 case nav2_costmap_2d::INSCRIBED_INFLATED_OBSTACLE: return 'I';
39 case nav2_costmap_2d::FREE_SPACE: return '.';
40 default: return '0' + (unsigned char) (10 * cost / 255);
41 }
42}

55unsigned int countValues(
56 nav2_costmap_2d::Costmap2D & costmap,
57 unsigned char value, bool equal = true)
58{
59 unsigned int count = 0;
60 for (unsigned int i = 0; i < costmap.getSizeInCellsY(); i++) {
61 for (unsigned int j = 0; j < costmap.getSizeInCellsX(); j++) {
62 unsigned char c = costmap.getCost(j, i);
63 if ((equal && c == value) || (!equal && c != value)) {
64 count += 1;
65 }
66 }
67 }
68 return count;
69}

104void addObservation(
105 std::shared_ptr<nav2_costmap_2d::ObstacleLayer> olayer, double x, double y, double z = 0.0,
106 double ox = 0.0, double oy = 0.0, double oz = MAX_Z, bool marking = true, bool clearing = true,
107 double raytrace_max_range = 100.0,
108 double raytrace_min_range = 0.0,
109 double obstacle_max_range = 100.0,
110 double obstacle_min_range = 0.0)
111{
112 sensor_msgs::msg::PointCloud2 cloud;
113 sensor_msgs::PointCloud2Modifier modifier(cloud);
114 modifier.setPointCloud2FieldsByString(1, "xyz");
115 modifier.resize(1);
116 sensor_msgs::PointCloud2Iterator<float> iter_x(cloud, "x");
117 sensor_msgs::PointCloud2Iterator<float> iter_y(cloud, "y");
118 sensor_msgs::PointCloud2Iterator<float> iter_z(cloud, "z");
119 *iter_x = x;
120 *iter_y = y;
121 *iter_z = z;
122
123 geometry_msgs::msg::Point p;
124 p.x = ox;
125 p.y = oy;
126 p.z = oz;
127
128 nav2_costmap_2d::Observation obs(p, cloud, obstacle_max_range, obstacle_min_range,
129 raytrace_max_range, raytrace_min_range);
130 olayer->addStaticObservation(obs, marking, clearing);
131}
