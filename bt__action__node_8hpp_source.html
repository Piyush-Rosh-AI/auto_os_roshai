92 void createActionClient(const std::string & action_name)
93 {
94 // Now that we have the ROS node to use, create the action client for this BT action
95 action_client_ = rclcpp_action::create_client<ActionT>(node_, action_name, callback_group_);
96
97 // Make sure the server is actually there before continuing
98 RCLCPP_DEBUG(node_->get_logger(), "Waiting for \"%s\" action server", action_name.c_str());
99 if (!action_client_->wait_for_action_server(wait_for_service_timeout_)) {
100 RCLCPP_ERROR(
101 node_->get_logger(), "\"%s\" action server not available after waiting for %.2fs",
102 action_name.c_str(),
103 wait_for_service_timeout_.count() / 1000.0);
104 throw std::runtime_error(
105 std::string("Action server ") + action_name +
106 std::string(" not available"));

190 BT::NodeStatus tick() override
191 {
192 // first step to be done only at the beginning of the Action
193 if (!BT::isStatusActive(status())) {
194 // setting the status to RUNNING to notify the BT Loggers (if any)
195 setStatus(BT::NodeStatus::RUNNING);
196
197 // reset the flag to send the goal or not, allowing the user the option to set it in on_tick
198 should_send_goal_ = true;
199
200 // user defined callback, may modify "should_send_goal_".
201 on_tick();
202
203 if (!should_send_goal_) {
204 return BT::NodeStatus::FAILURE;
205 }
206 send_new_goal();
207 }
208
209 try {
210 // if new goal was sent and action server has not yet responded
211 // check the future goal handle
212 if (future_goal_handle_) {
213 auto elapsed =
214 (node_->now() - time_goal_sent_).template to_chrono<std::chrono::milliseconds>();
215 if (!is_future_goal_handle_complete(elapsed)) {
216 // return RUNNING if there is still some time before timeout happens
217 if (elapsed < server_timeout_) {
218 return BT::NodeStatus::RUNNING;
219 }
220 // if server has taken more time than the specified timeout value return FAILURE
221 RCLCPP_WARN(
222 node_->get_logger(),
223 "Timed out while waiting for action server to acknowledge goal request for %s",
224 action_name_.c_str());
225 future_goal_handle_.reset();
226 return BT::NodeStatus::FAILURE;
227 }
228 }
229
230 // The following code corresponds to the "RUNNING" loop
231 if (rclcpp::ok() && !goal_result_available_) {
232 // user defined callback. May modify the value of "goal_updated_"
233 on_wait_for_result(feedback_);
234
235 // reset feedback to avoid stale information
236 feedback_.reset();
237
238 auto goal_status = goal_handle_->get_status();
239 if (goal_updated_ &&
240 (goal_status == action_msgs::msg::GoalStatus::STATUS_EXECUTING ||
241 goal_status == action_msgs::msg::GoalStatus::STATUS_ACCEPTED))
242 {
243 goal_updated_ = false;
244 send_new_goal();
245 auto elapsed =
246 (node_->now() - time_goal_sent_).template to_chrono<std::chrono::milliseconds>();
247 if (!is_future_goal_handle_complete(elapsed)) {
248 if (elapsed < server_timeout_) {
249 return BT::NodeStatus::RUNNING;
250 }
251 RCLCPP_WARN(
252 node_->get_logger(),
253 "Timed out while waiting for action server to acknowledge goal request for %s",
254 action_name_.c_str());
255 future_goal_handle_.reset();
256 return BT::NodeStatus::FAILURE;
257 }
258 }
259
260 callback_group_executor_.spin_some();
261
262 // check if, after invoking spin_some(), we finally received the result
263 if (!goal_result_available_) {
264 // Yield this Action, returning RUNNING
265 return BT::NodeStatus::RUNNING;
266 }
267 }
268 } catch (const std::runtime_error & e) {
269 if (e.what() == std::string("send_goal failed") ||
270 e.what() == std::string("Goal was rejected by the action server"))
271 {
272 // Action related failure that should not fail the tree, but the node
273 return BT::NodeStatus::FAILURE;
274 } else {
275 // Internal exception to propagate to the tree
276 throw e;
277 }
278 }
279
280 BT::NodeStatus status;
281 switch (result_.code) {
282 case rclcpp_action::ResultCode::SUCCEEDED:
283 status = on_success();
284 break;
285
286 case rclcpp_action::ResultCode::ABORTED:
287 status = on_aborted();
288 break;
289
290 case rclcpp_action::ResultCode::CANCELED:
291 status = on_cancelled();
292 break;
293
294 default:
295 throw std::logic_error("BtActionNode::Tick: invalid status value");
296 }
297
298 goal_handle_.reset();
299 return status;
300 }

306 void halt() override
307 {
308 if (should_cancel_goal()) {
309 auto future_result = action_client_->async_get_result(goal_handle_);
310 auto future_cancel = action_client_->async_cancel_goal(goal_handle_);
311 if (callback_group_executor_.spin_until_future_complete(future_cancel, server_timeout_) !=
312 rclcpp::FutureReturnCode::SUCCESS)
313 {
314 RCLCPP_ERROR(
315 node_->get_logger(),
316 "Failed to cancel action server for %s", action_name_.c_str());
317 }
318
319 if (callback_group_executor_.spin_until_future_complete(future_result, server_timeout_) !=
320 rclcpp::FutureReturnCode::SUCCESS)
321 {
322 RCLCPP_ERROR(
323 node_->get_logger(),
324 "Failed to get result for %s in node halt!", action_name_.c_str());
325 }
326
327 on_cancelled();
328 }
329
330 // this is probably redundant, since the parent node
331 // is supposed to call it, but we keep it, just in case
332 resetStatus();
333 }

335protected:
340 bool should_cancel_goal()
341 {
342 // Shut the node down if it is currently running
343 if (status() != BT::NodeStatus::RUNNING) {
344 return false;
345 }
346
347 // No need to cancel the goal if goal handle is invalid
348 if (!goal_handle_) {
349 return false;
350 }
351
352 callback_group_executor_.spin_some();
353 auto status = goal_handle_->get_status();
354
355 // Check if the goal is still executing
356 return status == action_msgs::msg::GoalStatus::STATUS_ACCEPTED ||
357 status == action_msgs::msg::GoalStatus::STATUS_EXECUTING;
358 }

363 void send_new_goal()
364 {
365 goal_result_available_ = false;
366 auto send_goal_options = typename rclcpp_action::Client<ActionT>::SendGoalOptions();
367 send_goal_options.result_callback =
368 [this](const typename rclcpp_action::ClientGoalHandle<ActionT>::WrappedResult & result) {
369 if (future_goal_handle_) {
370 RCLCPP_DEBUG(
371 node_->get_logger(),
372 "Goal result for %s available, but it hasn't received the goal response yet. "
373 "It's probably a goal result for the last goal request", action_name_.c_str());
374 return;
375 }
376
377 // TODO(#1652): a work around until rcl_action interface is updated
378 // if goal ids are not matched, the older goal call this callback so ignore the result
379 // if matched, it must be processed (including aborted)
380 if (this->goal_handle_->get_goal_id() == result.goal_id) {
381 goal_result_available_ = true;
382 result_ = result;
383 emitWakeUpSignal();
384 }
385 };
386 send_goal_options.feedback_callback =
387 [this](typename rclcpp_action::ClientGoalHandle<ActionT>::SharedPtr,
388 const std::shared_ptr<const typename ActionT::Feedback> feedback) {
389 feedback_ = feedback;
390 emitWakeUpSignal();
391 };
392
393 future_goal_handle_ = std::make_shared<
394 std::shared_future<typename rclcpp_action::ClientGoalHandle<ActionT>::SharedPtr>>(
395 action_client_->async_send_goal(goal_, send_goal_options));
396 time_goal_sent_ = node_->now();
397 }

405 bool is_future_goal_handle_complete(std::chrono::milliseconds & elapsed)
406 {
407 auto remaining = server_timeout_ - elapsed;
408
409 // server has already timed out, no need to sleep
410 if (remaining <= std::chrono::milliseconds(0)) {
411 future_goal_handle_.reset();
412 return false;
413 }
414
415 auto timeout = remaining > max_timeout_ ? max_timeout_ : remaining;
416 auto result =
417 callback_group_executor_.spin_until_future_complete(*future_goal_handle_, timeout);
418 elapsed += timeout;
419
420 if (result == rclcpp::FutureReturnCode::INTERRUPTED) {
421 future_goal_handle_.reset();
422 throw std::runtime_error("send_goal failed");
423 }
424
425 if (result == rclcpp::FutureReturnCode::SUCCESS) {
426 goal_handle_ = future_goal_handle_->get();
427 future_goal_handle_.reset();
428 if (!goal_handle_) {
429 throw std::runtime_error("Goal was rejected by the action server");
430 }
431 return true;
432 }
433
434 return false;
435 }
