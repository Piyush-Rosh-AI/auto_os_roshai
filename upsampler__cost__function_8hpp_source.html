99 virtual bool Evaluate(
100 const double * parameters,
101 double * cost,
102 double * gradient) const
103 {
104 Eigen::Vector2d xi;
105 Eigen::Vector2d xi_p1;
106 Eigen::Vector2d xi_m1;
107 uint x_index, y_index;
108 cost[0] = 0.0;
109 double cost_raw = 0.0;
110 double grad_x_raw = 0.0;
111 double grad_y_raw = 0.0;
112
113 // cache some computations between the residual and jacobian
114 CurvatureComputations curvature_params;
115
116 for (int i = 0; i != NumParameters() / 2; i++) {
117 x_index = 2 * i;
118 y_index = 2 * i + 1;
119 gradient[x_index] = 0.0;
120 gradient[y_index] = 0.0;
121 if (i < 1 || i >= NumParameters() / 2 - 1) {
122 continue;
123 }
124
125 // if original point's neighbors TODO
126 if (i % _upsample_ratio == 1) {
127 continue;
128 }
129
130 xi = Eigen::Vector2d(parameters[x_index], parameters[y_index]);
131
132 // TODO(stevemacenski): from deep copy to make sure no feedback _path
133 xi_p1 = _path.at(i + 1);
134 xi_m1 = _path.at(i - 1);
135 // xi_p1 = Eigen::Vector2d(parameters[x_index + 2], parameters[y_index + 2]);
136 // xi_m1 = Eigen::Vector2d(parameters[x_index - 2], parameters[y_index - 2]);
137
138 // compute cost
139 addSmoothingResidual(15000, xi, xi_p1, xi_m1, cost_raw);
140 addCurvatureResidual(60.0, xi, xi_p1, xi_m1, curvature_params, cost_raw);
141
142 if (gradient != NULL) {
143 // compute gradient
144 addSmoothingJacobian(15000, xi, xi_p1, xi_m1, grad_x_raw, grad_y_raw);
145 addCurvatureJacobian(60.0, xi, xi_p1, xi_m1, curvature_params, grad_x_raw, grad_y_raw);
146
147 gradient[x_index] = grad_x_raw;
148 gradient[y_index] = grad_y_raw;
149 }
150 }
151
152 cost[0] = cost_raw;
153 return true;
154 }

171 inline void addSmoothingResidual(
172 const double & weight,
173 const Eigen::Vector2d & pt,
174 const Eigen::Vector2d & pt_p,
175 const Eigen::Vector2d & pt_m,
176 double & r) const
177 {
178 r += weight * (
179 pt_p.dot(pt_p) -
180 4 * pt_p.dot(pt) +
181 2 * pt_p.dot(pt_m) +
182 4 * pt.dot(pt) -
183 4 * pt.dot(pt_m) +
184 pt_m.dot(pt_m)); // objective function value
185 }

196 inline void addSmoothingJacobian(
197 const double & weight,
198 const Eigen::Vector2d & pt,
199 const Eigen::Vector2d & pt_p,
200 const Eigen::Vector2d & pt_m,
201 double & j0,
202 double & j1) const
203 {
204 j0 += weight *
205 (-4 * pt_m[0] + 8 * pt[0] - 4 * pt_p[0]); // xi x component of partial-derivative
206 j1 += weight *
207 (-4 * pt_m[1] + 8 * pt[1] - 4 * pt_p[1]); // xi y component of partial-derivative
208 }

217 inline void getCurvatureParams(
218 const Eigen::Vector2d & pt,
219 const Eigen::Vector2d & pt_p,
220 const Eigen::Vector2d & pt_m,
221 CurvatureComputations & curvature_params) const
222 {
223 curvature_params.valid = true;
224 curvature_params.delta_xi = Eigen::Vector2d(pt[0] - pt_m[0], pt[1] - pt_m[1]);
225 curvature_params.delta_xi_p = Eigen::Vector2d(pt_p[0] - pt[0], pt_p[1] - pt[1]);
226 curvature_params.delta_xi_norm = curvature_params.delta_xi.norm();
227 curvature_params.delta_xi_p_norm = curvature_params.delta_xi_p.norm();
228
229 if (curvature_params.delta_xi_norm < EPSILON || curvature_params.delta_xi_p_norm < EPSILON ||
230 std::isnan(curvature_params.delta_xi_p_norm) || std::isnan(curvature_params.delta_xi_norm) ||
231 std::isinf(curvature_params.delta_xi_p_norm) || std::isinf(curvature_params.delta_xi_norm))
232 {
233 // ensure we have non-nan values returned
234 curvature_params.valid = false;
235 return;
236 }
237
238 const double & delta_xi_by_xi_p =
239 curvature_params.delta_xi_norm * curvature_params.delta_xi_p_norm;
240 double projection =
241 curvature_params.delta_xi.dot(curvature_params.delta_xi_p) / delta_xi_by_xi_p;
242 if (fabs(1 - projection) < EPSILON || fabs(projection + 1) < EPSILON) {
243 projection = 1.0;
244 }
245
246 curvature_params.delta_phi_i = std::acos(projection);
247 curvature_params.turning_rad = curvature_params.delta_phi_i / curvature_params.delta_xi_norm;
248
249 curvature_params.ki_minus_kmax = curvature_params.turning_rad - _upsample_ratio *
250 _params.max_curvature;
251 // TODO(stevemacenski) is use of upsample_ratio correct here? small number?
252 // TODO(stevemacenski) can remove the subtraction with a
253 // lower weight value, does have direction issue, maybe just tuning?
254
255 if (curvature_params.ki_minus_kmax <= EPSILON) {
256 // Quadratic penalty need not apply
257 curvature_params.valid = false;
258 }
259 }

270 inline void addCurvatureResidual(
271 const double & weight,
272 const Eigen::Vector2d & pt,
273 const Eigen::Vector2d & pt_p,
274 const Eigen::Vector2d & pt_m,
275 CurvatureComputations & curvature_params,
276 double & r) const
277 {
278 getCurvatureParams(pt, pt_p, pt_m, curvature_params);
279
280 if (!curvature_params.isValid()) {
281 return;
282 }
283
284 r += weight *
285 curvature_params.ki_minus_kmax * curvature_params.ki_minus_kmax; // objective function value
286 }

298 inline void addCurvatureJacobian(
299 const double & weight,
300 const Eigen::Vector2d & pt,
301 const Eigen::Vector2d & pt_p,
302 const Eigen::Vector2d & /*pt_m*/,
303 CurvatureComputations & curvature_params,
304 double & j0,
305 double & j1) const
306 {
307 if (!curvature_params.isValid()) {
308 return;
309 }
310
311 const double & partial_delta_phi_i_wrt_cost_delta_phi_i =
312 -1 / std::sqrt(1 - std::pow(std::cos(curvature_params.delta_phi_i), 2));
313 // const Eigen::Vector2d ones = Eigen::Vector2d(1.0, 1.0);
314 auto neg_pt_plus = -1 * pt_p;
315 Eigen::Vector2d p1 = normalizedOrthogonalComplement(
316 pt, neg_pt_plus, curvature_params.delta_xi_norm, curvature_params.delta_xi_p_norm);
317 Eigen::Vector2d p2 = normalizedOrthogonalComplement(
318 neg_pt_plus, pt, curvature_params.delta_xi_p_norm, curvature_params.delta_xi_norm);
319
320 const double & u = 2 * curvature_params.ki_minus_kmax;
321 const double & common_prefix =
322 (1 / curvature_params.delta_xi_norm) * partial_delta_phi_i_wrt_cost_delta_phi_i;
323 const double & common_suffix = curvature_params.delta_phi_i /
324 (curvature_params.delta_xi_norm * curvature_params.delta_xi_norm);
325 const Eigen::Vector2d & d_delta_xi_d_xi = curvature_params.delta_xi /
326 curvature_params.delta_xi_norm;
327
328 const Eigen::Vector2d jacobian = u *
329 (common_prefix * (-p1 - p2) - (common_suffix * d_delta_xi_d_xi));
330 const Eigen::Vector2d jacobian_im1 = u *
331 (common_prefix * p2 + (common_suffix * d_delta_xi_d_xi));
332 const Eigen::Vector2d jacobian_ip1 = u * (common_prefix * p1);
333 j0 += weight * jacobian[0]; // xi y component of partial-derivative
334 j1 += weight * jacobian[1]; // xi x component of partial-derivative
335 // j0 += weight *
336 // (jacobian_im1[0] + 2 * jacobian[0] + jacobian_ip1[0]);
337 // j1 += weight *
338 // (jacobian_im1[1] + 2 * jacobian[1] + jacobian_ip1[1]);
339 }

349 inline Eigen::Vector2d normalizedOrthogonalComplement(
350 const Eigen::Vector2d & a,
351 const Eigen::Vector2d & b,
352 const double & a_norm,
353 const double & b_norm) const
354 {
355 return (a - (a.dot(b) * b / b.squaredNorm())) / (a_norm * b_norm);
356 }
