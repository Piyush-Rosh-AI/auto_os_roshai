210inline bool withinPositionGoalTolerance(
211 nav2_core::GoalChecker * goal_checker,
212 const geometry_msgs::msg::Pose & robot,
213 const models::Path & path)
214{
215 const auto goal_idx = path.x.shape(0) - 1;
216 const auto goal_x = path.x(goal_idx);
217 const auto goal_y = path.y(goal_idx);
218
219 if (goal_checker) {
220 geometry_msgs::msg::Pose pose_tolerance;
221 geometry_msgs::msg::Twist velocity_tolerance;
222 goal_checker->getTolerances(pose_tolerance, velocity_tolerance);
223
224 const auto pose_tolerance_sq = pose_tolerance.position.x * pose_tolerance.position.x;
225
226 auto dx = robot.position.x - goal_x;
227 auto dy = robot.position.y - goal_y;
228
229 auto dist_sq = dx * dx + dy * dy;
230
231 if (dist_sq < pose_tolerance_sq) {
232 return true;
233 }
234 }
235
236 return false;
237}

276template<typename T>
277auto normalize_angles(const T & angles)
278{
279 auto theta = xt::eval(xt::fmod(angles + M_PIF, 2.0f * M_PIF));
280 return xt::eval(xt::where(theta < 0.0f, theta + M_PIF, theta - M_PIF));
281}

310inline size_t findPathFurthestReachedPoint(const CriticData & data)
311{
312 const auto traj_x = xt::view(data.trajectories.x, xt::all(), -1, xt::newaxis());
313 const auto traj_y = xt::view(data.trajectories.y, xt::all(), -1, xt::newaxis());
314
315 const auto dx = data.path.x - traj_x;
316 const auto dy = data.path.y - traj_y;
317
318 const auto dists = dx * dx + dy * dy;
319
320 size_t max_id_by_trajectories = 0, min_id_by_path = 0;
321 float min_distance_by_path = std::numeric_limits<float>::max();
322
323 for (size_t i = 0; i < dists.shape(0); i++) {
324 min_id_by_path = 0;
325 min_distance_by_path = std::numeric_limits<float>::max();
326 for (size_t j = max_id_by_trajectories; j < dists.shape(1); j++) {
327 const float cur_dist = dists(i, j);
328 if (cur_dist < min_distance_by_path) {
329 min_distance_by_path = cur_dist;
330 min_id_by_path = j;
331 }
332 }
333 max_id_by_trajectories = std::max(max_id_by_trajectories, min_id_by_path);
334 }
335 return max_id_by_trajectories;
336}

353inline void findPathCosts(
354 CriticData & data,
355 std::shared_ptr<nav2_costmap_2d::Costmap2DROS> costmap_ros)
356{
357 auto * costmap = costmap_ros->getCostmap();
358 unsigned int map_x, map_y;
359 const size_t path_segments_count = data.path.x.shape(0) - 1;
360 data.path_pts_valid = std::vector<bool>(path_segments_count, false);
361 const bool tracking_unknown = costmap_ros->getLayeredCostmap()->isTrackingUnknown();
362 for (unsigned int idx = 0; idx < path_segments_count; idx++) {
363 if (!costmap->worldToMap(data.path.x(idx), data.path.y(idx), map_x, map_y)) {
364 (*data.path_pts_valid)[idx] = false;
365 continue;
366 }
367
368 switch (costmap->getCost(map_x, map_y)) {
369 case (nav2_costmap_2d::LETHAL_OBSTACLE):
370 (*data.path_pts_valid)[idx] = false;
371 continue;
372 case (nav2_costmap_2d::INSCRIBED_INFLATED_OBSTACLE):
373 (*data.path_pts_valid)[idx] = false;
374 continue;
375 case (nav2_costmap_2d::NO_INFORMATION):
376 (*data.path_pts_valid)[idx] = tracking_unknown ? true : false;
377 continue;
378 }
379
380 (*data.path_pts_valid)[idx] = true;
381 }
382}

405inline float posePointAngle(
406 const geometry_msgs::msg::Pose & pose, double point_x, double point_y, bool forward_preference)
407{
408 float pose_x = pose.position.x;
409 float pose_y = pose.position.y;
410 float pose_yaw = tf2::getYaw(pose.orientation);
411
412 float yaw = atan2f(point_y - pose_y, point_x - pose_x);
413
414 // If no preference for forward, return smallest angle either in heading or 180 of heading
415 if (!forward_preference) {
416 return std::min(
417 fabs(angles::shortest_angular_distance(yaw, pose_yaw)),
418 fabs(angles::shortest_angular_distance(yaw, angles::normalize_angle(pose_yaw + M_PIF))));
419 }
420
421 return fabs(angles::shortest_angular_distance(yaw, pose_yaw));
422}

432inline float posePointAngle(
433 const geometry_msgs::msg::Pose & pose,
434 double point_x, double point_y, double point_yaw)
435{
436 float pose_x = static_cast<float>(pose.position.x);
437 float pose_y = static_cast<float>(pose.position.y);
438 float pose_yaw = static_cast<float>(tf2::getYaw(pose.orientation));
439
440 float yaw = atan2f(static_cast<float>(point_y) - pose_y, static_cast<float>(point_x) - pose_x);
441
442 if (fabs(angles::shortest_angular_distance(yaw, static_cast<float>(point_yaw))) > M_PIF_2) {
443 yaw = angles::normalize_angle(yaw + M_PIF);
444 }
445
446 return fabs(angles::shortest_angular_distance(yaw, pose_yaw));
447}

455inline void savitskyGolayFilter(
456 models::ControlSequence & control_sequence,
457 std::array<mppi::models::Control, 4> & control_history,
458 const models::OptimizerSettings & settings)
459{
460 // Savitzky-Golay Quadratic, 9-point Coefficients
461 xt::xarray<float> filter = {-21.0, 14.0, 39.0, 54.0, 59.0, 54.0, 39.0, 14.0, -21.0};
462 filter /= 231.0;
463
464 const unsigned int num_sequences = control_sequence.vx.shape(0) - 1;
465
466 // Too short to smooth meaningfully
467 if (num_sequences < 20) {
468 return;
469 }
470
471 auto applyFilter = [&](const xt::xarray<float> & data) -> float {
472 return xt::sum(data * filter, {0}, immediate)();
473 };
474
475 auto applyFilterOverAxis =
476 [&](xt::xtensor<float, 1> & sequence,
477 const float hist_0, const float hist_1, const float hist_2, const float hist_3) -> void
478 {
479 unsigned int idx = 0;
480 sequence(idx) = applyFilter(
481 {
482 hist_0,
483 hist_1,
484 hist_2,
485 hist_3,
486 sequence(idx),
487 sequence(idx + 1),
488 sequence(idx + 2),
489 sequence(idx + 3),
490 sequence(idx + 4)});
491
492 idx++;
493 sequence(idx) = applyFilter(
494 {
495 hist_1,
496 hist_2,
497 hist_3,
498 sequence(idx - 1),
499 sequence(idx),
500 sequence(idx + 1),
501 sequence(idx + 2),
502 sequence(idx + 3),
503 sequence(idx + 4)});
504
505 idx++;
506 sequence(idx) = applyFilter(
507 {
508 hist_2,
509 hist_3,
510 sequence(idx - 2),
511 sequence(idx - 1),
512 sequence(idx),
513 sequence(idx + 1),
514 sequence(idx + 2),
515 sequence(idx + 3),
516 sequence(idx + 4)});
517
518 idx++;
519 sequence(idx) = applyFilter(
520 {
521 hist_3,
522 sequence(idx - 3),
523 sequence(idx - 2),
524 sequence(idx - 1),
525 sequence(idx),
526 sequence(idx + 1),
527 sequence(idx + 2),
528 sequence(idx + 3),
529 sequence(idx + 4)});
530
531 for (idx = 4; idx != num_sequences - 4; idx++) {
532 sequence(idx) = applyFilter(
533 {
534 sequence(idx - 4),
535 sequence(idx - 3),
536 sequence(idx - 2),
537 sequence(idx - 1),
538 sequence(idx),
539 sequence(idx + 1),
540 sequence(idx + 2),
541 sequence(idx + 3),
542 sequence(idx + 4)});
543 }
544
545 idx++;
546 sequence(idx) = applyFilter(
547 {
548 sequence(idx - 4),
549 sequence(idx - 3),
550 sequence(idx - 2),
551 sequence(idx - 1),
552 sequence(idx),
553 sequence(idx + 1),
554 sequence(idx + 2),
555 sequence(idx + 3),
556 sequence(idx + 3)});
557
558 idx++;
559 sequence(idx) = applyFilter(
560 {
561 sequence(idx - 4),
562 sequence(idx - 3),
563 sequence(idx - 2),
564 sequence(idx - 1),
565 sequence(idx),
566 sequence(idx + 1),
567 sequence(idx + 2),
568 sequence(idx + 2),
569 sequence(idx + 2)});
570
571 idx++;
572 sequence(idx) = applyFilter(
573 {
574 sequence(idx - 4),
575 sequence(idx - 3),
576 sequence(idx - 2),
577 sequence(idx - 1),
578 sequence(idx),
579 sequence(idx + 1),
580 sequence(idx + 1),
581 sequence(idx + 1),
582 sequence(idx + 1)});
583
584 idx++;
585 sequence(idx) = applyFilter(
586 {
587 sequence(idx - 4),
588 sequence(idx - 3),
589 sequence(idx - 2),
590 sequence(idx - 1),
591 sequence(idx),
592 sequence(idx),
593 sequence(idx),
594 sequence(idx),
595 sequence(idx)});
596 };
597
598 // Filter trajectories
599 applyFilterOverAxis(
600 control_sequence.vx, control_history[0].vx,
601 control_history[1].vx, control_history[2].vx, control_history[3].vx);
602 applyFilterOverAxis(
603 control_sequence.vy, control_history[0].vy,
604 control_history[1].vy, control_history[2].vy, control_history[3].vy);
605 applyFilterOverAxis(
606 control_sequence.wz, control_history[0].wz,
607 control_history[1].wz, control_history[2].wz, control_history[3].wz);
608
609 // Update control history
610 unsigned int offset = settings.shift_control_sequence ? 1 : 0;
611 control_history[0] = control_history[1];
612 control_history[1] = control_history[2];
613 control_history[2] = control_history[3];
614 control_history[3] = {
615 control_sequence.vx(offset),
616 control_sequence.vy(offset),
617 control_sequence.wz(offset)};
618}

625inline unsigned int findFirstPathInversion(nav_msgs::msg::Path & path)
626{
627 // At least 3 poses for a possible inversion
628 if (path.poses.size() < 3) {
629 return path.poses.size();
630 }
631
632 // Iterating through the path to determine the position of the path inversion
633 for (unsigned int idx = 1; idx < path.poses.size() - 1; ++idx) {
634 // We have two vectors for the dot product OA and AB. Determining the vectors.
635 float oa_x = path.poses[idx].pose.position.x -
636 path.poses[idx - 1].pose.position.x;
637 float oa_y = path.poses[idx].pose.position.y -
638 path.poses[idx - 1].pose.position.y;
639 float ab_x = path.poses[idx + 1].pose.position.x -
640 path.poses[idx].pose.position.x;
641 float ab_y = path.poses[idx + 1].pose.position.y -
642 path.poses[idx].pose.position.y;
643
644 // Checking for the existance of cusp, in the path, using the dot product.
645 float dot_product = (oa_x * ab_x) + (oa_y * ab_y);
646 if (dot_product < 0.0f) {
647 return idx + 1;
648 }
649 }
650
651 return path.poses.size();
652}

659inline unsigned int removePosesAfterFirstInversion(nav_msgs::msg::Path & path)
660{
661 nav_msgs::msg::Path cropped_path = path;
662 const unsigned int first_after_inversion = findFirstPathInversion(cropped_path);
663 if (first_after_inversion == path.poses.size()) {
664 return 0u;
665 }
666
667 cropped_path.poses.erase(
668 cropped_path.poses.begin() + first_after_inversion, cropped_path.poses.end());
669 path = cropped_path;
670 return first_after_inversion;
671}

679inline unsigned int findClosestPathPt(
680 const std::vector<float> & vec, const float dist, const unsigned int init = 0u)
681{
682 float distim1 = init != 0u ? vec[init] : 0.0f; // First is 0, no accumulated distance yet
683 float disti = 0.0f;
684 const unsigned int size = vec.size();
685 for (unsigned int i = init + 1; i != size; i++) {
686 disti = vec[i];
687 if (disti > dist) {
688 if (i > 0 && dist - distim1 < disti - dist) {
689 return i - 1;
690 }
691 return i;
692 }
693 distim1 = disti;
694 }
695 return size - 1;
696}
