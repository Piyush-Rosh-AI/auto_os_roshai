117 virtual bool Evaluate(
118 const double * parameters,
119 double * cost,
120 double * gradient) const
121 {
122 Eigen::Vector2d xi;
123 Eigen::Vector2d xi_p1;
124 Eigen::Vector2d xi_m1;
125 unsigned int x_index, y_index;
126 cost[0] = 0.0;
127 double cost_raw = 0.0;
128 double grad_x_raw = 0.0;
129 double grad_y_raw = 0.0;
130 unsigned int mx, my;
131 bool valid_coords = true;
132 double costmap_cost = 0.0;
133
134 // cache some computations between the residual and jacobian
135 CurvatureComputations curvature_params;
136
137 for (int i = 0; i != NumParameters() / 2; i++) {
138 x_index = 2 * i;
139 y_index = 2 * i + 1;
140 gradient[x_index] = 0.0;
141 gradient[y_index] = 0.0;
142 if (i < 1 || i >= NumParameters() / 2 - 1) {
143 continue;
144 }
145
146 xi = Eigen::Vector2d(parameters[x_index], parameters[y_index]);
147 xi_p1 = Eigen::Vector2d(parameters[x_index + 2], parameters[y_index + 2]);
148 xi_m1 = Eigen::Vector2d(parameters[x_index - 2], parameters[y_index - 2]);
149
150 // compute cost
151 addSmoothingResidual(_params.smooth_weight, xi, xi_p1, xi_m1, cost_raw);
152 addCurvatureResidual(_params.curvature_weight, xi, xi_p1, xi_m1, curvature_params, cost_raw);
153 addDistanceResidual(_params.distance_weight, xi, _original_path->at(i), cost_raw);
154
155 if (valid_coords = _costmap->worldToMap(xi[0], xi[1], mx, my)) {
156 costmap_cost = _costmap->getCost(mx, my);
157 addCostResidual(_params.costmap_weight, costmap_cost, cost_raw, xi);
158 }
159
160 if (gradient != NULL) {
161 // compute gradient
162 gradient[x_index] = 0.0;
163 gradient[y_index] = 0.0;
164 addSmoothingJacobian(_params.smooth_weight, xi, xi_p1, xi_m1, grad_x_raw, grad_y_raw);
165 addCurvatureJacobian(
166 _params.curvature_weight, xi, xi_p1, xi_m1, curvature_params,
167 grad_x_raw, grad_y_raw);
168 addDistanceJacobian(
169 _params.distance_weight, xi, _original_path->at(
170 i), grad_x_raw, grad_y_raw);
171
172 if (valid_coords) {
173 addCostJacobian(_params.costmap_weight, mx, my, costmap_cost, grad_x_raw, grad_y_raw);
174 }
175
176 gradient[x_index] = grad_x_raw;
177 gradient[y_index] = grad_y_raw;
178 }
179 }
180
181 cost[0] = cost_raw;
182
183 return true;
184 }

201 inline void addSmoothingResidual(
202 const double & weight,
203 const Eigen::Vector2d & pt,
204 const Eigen::Vector2d & pt_p,
205 const Eigen::Vector2d & pt_m,
206 double & r) const
207 {
208 r += weight * (
209 pt_p.dot(pt_p) -
210 4 * pt_p.dot(pt) +
211 2 * pt_p.dot(pt_m) +
212 4 * pt.dot(pt) -
213 4 * pt.dot(pt_m) +
214 pt_m.dot(pt_m)); // objective function value
215 }

226 inline void addSmoothingJacobian(
227 const double & weight,
228 const Eigen::Vector2d & pt,
229 const Eigen::Vector2d & pt_p,
230 const Eigen::Vector2d & pt_m,
231 double & j0,
232 double & j1) const
233 {
234 j0 += weight *
235 (-4 * pt_m[0] + 8 * pt[0] - 4 * pt_p[0]); // xi x component of partial-derivative
236 j1 += weight *
237 (-4 * pt_m[1] + 8 * pt[1] - 4 * pt_p[1]); // xi y component of partial-derivative
238 }

249 inline void addCurvatureResidual(
250 const double & weight,
251 const Eigen::Vector2d & pt,
252 const Eigen::Vector2d & pt_p,
253 const Eigen::Vector2d & pt_m,
254 CurvatureComputations & curvature_params,
255 double & r) const
256 {
257 curvature_params.valid = true;
258 curvature_params.delta_xi = Eigen::Vector2d(pt[0] - pt_m[0], pt[1] - pt_m[1]);
259 curvature_params.delta_xi_p = Eigen::Vector2d(pt_p[0] - pt[0], pt_p[1] - pt[1]);
260 curvature_params.delta_xi_norm = curvature_params.delta_xi.norm();
261 curvature_params.delta_xi_p_norm = curvature_params.delta_xi_p.norm();
262
263 if (curvature_params.delta_xi_norm < EPSILON || curvature_params.delta_xi_p_norm < EPSILON ||
264 std::isnan(curvature_params.delta_xi_p_norm) || std::isnan(curvature_params.delta_xi_norm) ||
265 std::isinf(curvature_params.delta_xi_p_norm) || std::isinf(curvature_params.delta_xi_norm))
266 {
267 // ensure we have non-nan values returned
268 curvature_params.valid = false;
269 return;
270 }
271
272 const double & delta_xi_by_xi_p =
273 curvature_params.delta_xi_norm * curvature_params.delta_xi_p_norm;
274 double projection =
275 curvature_params.delta_xi.dot(curvature_params.delta_xi_p) / delta_xi_by_xi_p;
276 if (fabs(1 - projection) < EPSILON || fabs(projection + 1) < EPSILON) {
277 projection = 1.0;
278 }
279
280 curvature_params.delta_phi_i = std::acos(projection);
281 curvature_params.turning_rad = curvature_params.delta_phi_i / curvature_params.delta_xi_norm;
282
283 curvature_params.ki_minus_kmax = curvature_params.turning_rad - _params.max_curvature;
284
285 if (curvature_params.ki_minus_kmax <= EPSILON) {
286 // Quadratic penalty need not apply
287 curvature_params.valid = false;
288 return;
289 }
290
291 r += weight *
292 curvature_params.ki_minus_kmax * curvature_params.ki_minus_kmax; // objective function value
293 }

305 inline void addCurvatureJacobian(
306 const double & weight,
307 const Eigen::Vector2d & pt,
308 const Eigen::Vector2d & pt_p,
309 const Eigen::Vector2d & /*pt_m*/,
310 CurvatureComputations & curvature_params,
311 double & j0,
312 double & j1) const
313 {
314 if (!curvature_params.isValid()) {
315 return;
316 }
317
318 const double & partial_delta_phi_i_wrt_cost_delta_phi_i =
319 -1 / std::sqrt(1 - std::pow(std::cos(curvature_params.delta_phi_i), 2));
320 // const Eigen::Vector2d ones = Eigen::Vector2d(1.0, 1.0);
321 auto neg_pt_plus = -1 * pt_p;
322 Eigen::Vector2d p1 = normalizedOrthogonalComplement(
323 pt, neg_pt_plus, curvature_params.delta_xi_norm, curvature_params.delta_xi_p_norm);
324 Eigen::Vector2d p2 = normalizedOrthogonalComplement(
325 neg_pt_plus, pt, curvature_params.delta_xi_p_norm, curvature_params.delta_xi_norm);
326
327 const double & u = 2 * curvature_params.ki_minus_kmax;
328 const double & common_prefix =
329 (1 / curvature_params.delta_xi_norm) * partial_delta_phi_i_wrt_cost_delta_phi_i;
330 const double & common_suffix = curvature_params.delta_phi_i /
331 (curvature_params.delta_xi_norm * curvature_params.delta_xi_norm);
332
333 const Eigen::Vector2d & d_delta_xi_d_xi = curvature_params.delta_xi /
334 curvature_params.delta_xi_norm;
335
336 const Eigen::Vector2d jacobian = u *
337 (common_prefix * (-p1 - p2) - (common_suffix * d_delta_xi_d_xi));
338 const Eigen::Vector2d jacobian_im1 = u *
339 (common_prefix * p2 + (common_suffix * d_delta_xi_d_xi));
340 const Eigen::Vector2d jacobian_ip1 = u * (common_prefix * p1);
341
342 // Old formulation we may require again.
343 // j0 += weight *
344 // (jacobian_im1[0] + 2 * jacobian[0] + jacobian_ip1[0]);
345 // j1 += weight *
346 // (jacobian_im1[1] + 2 * jacobian[1] + jacobian_ip1[1]);
347
348 j0 += weight * jacobian[0]; // xi x component of partial-derivative
349 j1 += weight * jacobian[1]; // xi x component of partial-derivative
350 }

427 inline void addCostJacobian(
428 const double & weight,
429 const unsigned int & mx,
430 const unsigned int & my,
431 const double & value,

440 const double common_prefix = -2.0 * _params.costmap_factor * weight * value * value;

443 j1 += common_prefix * grad[1]; // xi y component of partial-derivative

453 inline Eigen::Vector2d getCostmapGradient(
454 const unsigned int mx,
455 const unsigned int my) const
456 {
457 // find unit vector that describes that direction
458 // via 7 point taylor series approximation for gradient at Xi
459 Eigen::Vector2d gradient;
460
461 double l_1 = 0.0;
462 double l_2 = 0.0;
463 double l_3 = 0.0;
464 double r_1 = 0.0;
465 double r_2 = 0.0;
466 double r_3 = 0.0;
467
468 if (mx < _costmap->getSizeInCellsX()) {
469 r_1 = static_cast<double>(_costmap->getCost(mx + 1, my));
470 }
471 if (mx + 1 < _costmap->getSizeInCellsX()) {
472 r_2 = static_cast<double>(_costmap->getCost(mx + 2, my));
473 }
474 if (mx + 2 < _costmap->getSizeInCellsX()) {
475 r_3 = static_cast<double>(_costmap->getCost(mx + 3, my));
476 }
477
478 if (mx > 0) {
479 l_1 = static_cast<double>(_costmap->getCost(mx - 1, my));
480 }
481 if (mx - 1 > 0) {
482 l_2 = static_cast<double>(_costmap->getCost(mx - 2, my));
483 }
484 if (mx - 2 > 0) {
485 l_3 = static_cast<double>(_costmap->getCost(mx - 3, my));
486 }
487
488 gradient[1] = (45 * r_1 - 9 * r_2 + r_3 - 45 * l_1 + 9 * l_2 - l_3) / 60;
489
490 if (my < _costmap->getSizeInCellsY()) {
491 r_1 = static_cast<double>(_costmap->getCost(mx, my + 1));
492 }
493 if (my + 1 < _costmap->getSizeInCellsY()) {
494 r_2 = static_cast<double>(_costmap->getCost(mx, my + 2));
495 }
496 if (my + 2 < _costmap->getSizeInCellsY()) {
497 r_3 = static_cast<double>(_costmap->getCost(mx, my + 3));
498 }
499
500 if (my > 0) {
501 l_1 = static_cast<double>(_costmap->getCost(mx, my - 1));
502 }
503 if (my - 1 > 0) {
504 l_2 = static_cast<double>(_costmap->getCost(mx, my - 2));
505 }
506 if (my - 2 > 0) {
507 l_3 = static_cast<double>(_costmap->getCost(mx, my - 3));
508 }
509
510 gradient[0] = (45 * r_1 - 9 * r_2 + r_3 - 45 * l_1 + 9 * l_2 - l_3) / 60;
511
512 gradient.normalize();
513 return gradient;
514 }

525 const Eigen::Vector2d & a,
