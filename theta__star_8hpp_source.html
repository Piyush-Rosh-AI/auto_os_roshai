198 bool isSafe(const int & cx, const int & cy, double & cost) const
199 {
200 double curr_cost = getCost(cx, cy);
201 if ((costmap_->getCost(cx, cy) == UNKNOWN_COST && allow_unknown_) || curr_cost < LETHAL_COST) {
202 if (costmap_->getCost(cx, cy) == UNKNOWN_COST) {
203 curr_cost = OBS_COST - 1;
204 }
205 cost += w_traversal_cost_ * curr_cost * curr_cost / LETHAL_COST / LETHAL_COST;
206 return true;
207 } else {
208 return false;
209 }
210 }
