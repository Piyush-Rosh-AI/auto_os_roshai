101 bool smooth(
102 std::vector<Eigen::Vector2d> & path,
103 nav2_costmap_2d::Costmap2D * costmap,
104 const SmootherParams & params)
105 {
106 _options.max_solver_time_in_seconds = params.max_time;
107
108#ifdef _MSC_VER
109 std::vector<double> parameters_vec(path.size() * 2);
110 double * parameters = parameters_vec.data();
111#else
112 double parameters[path.size() * 2]; // NOLINT
113#endif
114 for (unsigned int i = 0; i != path.size(); i++) {
115 parameters[2 * i] = path[i][0];
116 parameters[2 * i + 1] = path[i][1];
117 }
118
119 ceres::GradientProblemSolver::Summary summary;
120 ceres::GradientProblem problem(new UnconstrainedSmootherCostFunction(&path, costmap, params));
121 ceres::Solve(_options, problem, parameters, &summary);
122
123 if (_debug) {
124 std::cout << summary.FullReport() << '\n';
125 }
126
127 if (!summary.IsSolutionUsable() || summary.initial_cost - summary.final_cost <= 0.0) {
128 return false;
129 }
130
131 for (unsigned int i = 0; i != path.size(); i++) {
132 path[i][0] = parameters[2 * i];
133 path[i][1] = parameters[2 * i + 1];
134 }
135
136 return true;
137 }
138
139private:
140 bool _debug;
141 ceres::GradientProblemSolver::Options _options;
