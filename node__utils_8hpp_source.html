84/* Declares static ROS2 parameter and sets it to a given value
85 * if it was not already declared.
86 *
87 * \param[in] node A node in which given parameter to be declared
88 * \param[in] param_name The name of parameter
89 * \param[in] default_value Parameter value to initialize with
90 * \param[in] parameter_descriptor Parameter descriptor (optional)
91 */
92template<typename NodeT>
93void declare_parameter_if_not_declared(
94 NodeT node,
95 const std::string & param_name,
96 const rclcpp::ParameterValue & default_value,
97 const rcl_interfaces::msg::ParameterDescriptor & parameter_descriptor =
98 rcl_interfaces::msg::ParameterDescriptor())
99{
100 if (!node->has_parameter(param_name)) {
101 node->declare_parameter(param_name, default_value, parameter_descriptor);
102 }
103}

106/* Declares static ROS2 parameter with given type if it was not already declared.
107 *
108 * \param[in] node A node in which given parameter to be declared
109 * \param[in] param_type The type of parameter
110 * \param[in] default_value Parameter value to initialize with
111 * \param[in] parameter_descriptor Parameter descriptor (optional)
112 */
113template<typename NodeT>
114void declare_parameter_if_not_declared(
115 NodeT node,
116 const std::string & param_name,
117 const rclcpp::ParameterType & param_type,
118 const rcl_interfaces::msg::ParameterDescriptor & parameter_descriptor =
119 rcl_interfaces::msg::ParameterDescriptor())
120{
121 if (!node->has_parameter(param_name)) {
122 node->declare_parameter(param_name, param_type, parameter_descriptor);
123 }
124}

135template<typename NodeT>
136std::string get_plugin_type_param(
137 NodeT node,
138 const std::string & plugin_name)
139{
140 declare_parameter_if_not_declared(node, plugin_name + ".plugin", rclcpp::PARAMETER_STRING);
141 std::string plugin_type;
142 try {
143 if (!node->get_parameter(plugin_name + ".plugin", plugin_type)) {
144 RCLCPP_FATAL(
145 node->get_logger(), "Can not get 'plugin' param value for %s", plugin_name.c_str());
146 throw std::runtime_error("No 'plugin' param for param ns!");
147 }
148 } catch (rclcpp::exceptions::ParameterUninitializedException & ex) {
149 RCLCPP_FATAL(node->get_logger(), "'plugin' param not defined for %s", plugin_name.c_str());
150 throw std::runtime_error("No 'plugin' param for param ns!");
151 }
152
153 return plugin_type;
154}
