28template<class T>
29Image<T> makeImage(size_t rows, size_t columns, std::vector<T> & buffer, size_t step = 0)
30{
31 step = std::max(step, columns);
32 buffer.resize(rows * step);
33 return Image<T>(rows, columns, buffer.data(), step);
34}

36template<class T>
37Image<T> clone(const Image<T> & source, std::vector<T> & buffer)
38{
39 buffer.resize(source.rows() * source.columns());
40 Image<T> result(source.rows(), source.columns(), buffer.data(), source.columns());
41
42 for (size_t row = 0; row < source.rows(); ++row) {
43 for (size_t column = 0; column < source.columns(); ++column) {
44 result.row(row)[column] = source.row(row)[column];

46 }

64template<class T>
65Image<T> imageFromString(
66 const std::string & s, std::vector<T> & buffer,
67 const std::map<char, T> & codes = {{'.', 0}, {'x', 255}})
68{
69 const size_t side_size = static_cast<size_t>(std::sqrt(s.size()));
70
71 if (size_t(side_size) * side_size != s.size()) {
72 throw std::logic_error("Test data error: parseBinaryMatrix: Unexpected input string size");
73 }
74
75 const size_t step = static_cast<size_t>(side_size * 3);
76
77 Image<T> image = makeImage(side_size, side_size, buffer, step);
78 auto iter = s.begin();
79 image.forEach(
80 [&](T & pixel) {
81 try {
82 pixel = codes.at(*iter);
83 ++iter;
84 } catch (...) {
85 throw std::logic_error(
86 "Test data error: parseBinaryMatrix: Unexpected symbol: " +
87 std::string(1, *iter));
88 }
89 });
90 return image;
91}

98inline bool isEqual(const Image<uint8_t> & a, const Image<uint8_t> & b)
99{
100 bool equal = a.rows() == b.rows() && a.columns() == b.columns();
101
102 for (size_t row = 0; row < a.rows() && equal; ++row) {
103 for (size_t column = 0; column < a.columns() && equal; ++column) {
104 equal = a.row(row)[column] == b.row(row)[column];
105 }
106 }
107 return equal;
108}
