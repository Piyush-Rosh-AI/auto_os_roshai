440 template<class ActionType>
441 inline void raytraceLine(
442 ActionType at, unsigned int x0, unsigned int y0, unsigned int x1,
443 unsigned int y1,
444 unsigned int max_length = UINT_MAX, unsigned int min_length = 0)
445 {
446 int dx_full = x1 - x0;
447 int dy_full = y1 - y0;
448
449 // we need to chose how much to scale our dominant dimension,
450 // based on the maximum length of the line
451 double dist = std::hypot(dx_full, dy_full);
452 if (dist < min_length) {
453 return;
454 }
455
456 unsigned int min_x0, min_y0;
457 if (dist > 0.0) {
458 // Adjust starting point and offset to start from min_length distance
459 min_x0 = (unsigned int)(x0 + dx_full / dist * min_length);
460 min_y0 = (unsigned int)(y0 + dy_full / dist * min_length);
461 } else {
462 // dist can be 0 if [x0, y0]==[x1, y1].
463 // In this case only this cell should be processed.
464 min_x0 = x0;
465 min_y0 = y0;
466 }
467 unsigned int offset = min_y0 * size_x_ + min_x0;
468
469 int dx = x1 - min_x0;
470 int dy = y1 - min_y0;
471
472 unsigned int abs_dx = abs(dx);
473 unsigned int abs_dy = abs(dy);
474
475 int offset_dx = sign(dx);
476 int offset_dy = sign(dy) * size_x_;
477
478 double scale = (dist == 0.0) ? 1.0 : std::min(1.0, max_length / dist);
479 // if x is dominant
480 if (abs_dx >= abs_dy) {
481 int error_y = abs_dx / 2;
482
483 bresenham2D(
484 at, abs_dx, abs_dy, error_y, offset_dx, offset_dy, offset, (unsigned int)(scale * abs_dx));
485 return;
486 }
487
488 // otherwise y is dominant
489 int error_x = abs_dy / 2;
490
491 bresenham2D(
492 at, abs_dy, abs_dx, error_x, offset_dy, offset_dx, offset, (unsigned int)(scale * abs_dy));
493 }
494
495private:
500 template<class ActionType>
501 inline void bresenham2D(
502 ActionType at, unsigned int abs_da, unsigned int abs_db, int error_b,
503 int offset_a,
504 int offset_b, unsigned int offset,
505 unsigned int max_length)
506 {
507 unsigned int end = std::min(max_length, abs_da);
508 for (unsigned int i = 0; i < end; ++i) {
509 at(offset);
510 offset += offset_a;
511 error_b += abs_db;
512 if ((unsigned int)error_b >= abs_da) {
513 offset += offset_b;
514 error_b -= abs_da;
515 }
516 }
517 at(offset);
518 }
