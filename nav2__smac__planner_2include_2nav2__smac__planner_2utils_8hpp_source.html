43inline geometry_msgs::msg::Pose getWorldCoords(
44 const float & mx, const float & my, const nav2_costmap_2d::Costmap2D * costmap)
45{
46 geometry_msgs::msg::Pose msg;
47 msg.position.x =
48 static_cast<float>(costmap->getOriginX()) + (mx - 0.5) * costmap->getResolution();

50 static_cast<float>(costmap->getOriginY()) + (my - 0.5) * costmap->getResolution();

75inline double findCircumscribedCost(std::shared_ptr<nav2_costmap_2d::Costmap2DROS> costmap)
76{
77 double result = -1.0;
78 std::vector<std::shared_ptr<nav2_costmap_2d::Layer>>::iterator layer;
79
80 // check if the costmap has an inflation layer
81 const auto inflation_layer = nav2_costmap_2d::InflationLayer::getInflationLayer(costmap);
82 if (inflation_layer != nullptr) {
83 double circum_radius = costmap->getLayeredCostmap()->getCircumscribedRadius();
84 double resolution = costmap->getCostmap()->getResolution();
85 result = static_cast<double>(inflation_layer->computeCost(circum_radius / resolution));
86 } else {
87 RCLCPP_WARN(
88 rclcpp::get_logger("computeCircumscribedCost"),
89 "No inflation layer found in costmap configuration. "
90 "If this is an SE2-collision checking plugin, it cannot use costmap potential "
91 "field to speed up collision checking by only checking the full footprint "
92 "when robot is within possibly-inscribed radius of an obstacle. This may "
93 "significantly slow down planning times!");
94 }
95
96 return result;
97}

144 MotionPose pose;
145 fromJsonToPose(json["poses"][i], pose);
146 motion_primitive.poses.push_back(pose);

155inline std::vector<geometry_msgs::msg::Point> transformFootprintToEdges(
156 const geometry_msgs::msg::Pose & pose,
157 const std::vector<geometry_msgs::msg::Point> & footprint)
158{
159 const double & x = pose.position.x;
160 const double & y = pose.position.y;
161 const double & yaw = tf2::getYaw(pose.orientation);
162
163 std::vector<geometry_msgs::msg::Point> out_footprint;
164 out_footprint.resize(2 * footprint.size());
165 for (unsigned int i = 0; i < footprint.size(); i++) {
166 out_footprint[2 * i].x = x + cos(yaw) * footprint[i].x - sin(yaw) * footprint[i].y;
167 out_footprint[2 * i].y = y + sin(yaw) * footprint[i].x + cos(yaw) * footprint[i].y;
168 if (i == 0) {
169 out_footprint.back().x = out_footprint[i].x;
170 out_footprint.back().y = out_footprint[i].y;
171 } else {
172 out_footprint[2 * i - 1].x = out_footprint[2 * i].x;
173 out_footprint[2 * i - 1].y = out_footprint[2 * i].y;
174 }
175 }
176 return out_footprint;
177}

187inline visualization_msgs::msg::Marker createMarker(
188 const std::vector<geometry_msgs::msg::Point> edge,
189 unsigned int i, const std::string & frame_id, const rclcpp::Time & timestamp)
190{
191 visualization_msgs::msg::Marker marker;
192 marker.header.frame_id = frame_id;
193 marker.header.stamp = timestamp;
194 marker.frame_locked = false;
195 marker.ns = "planned_footprint";
196 marker.action = visualization_msgs::msg::Marker::ADD;
197 marker.type = visualization_msgs::msg::Marker::LINE_LIST;
198 marker.lifetime = rclcpp::Duration(0, 0);
199
200 marker.id = i;
201 for (auto & point : edge) {
202 marker.points.push_back(point);
203 }
204
205 marker.pose.orientation.x = 0.0;
206 marker.pose.orientation.y = 0.0;
207 marker.pose.orientation.z = 0.0;
208 marker.pose.orientation.w = 1.0;
209 marker.scale.x = 0.05;
210 marker.scale.y = 0.05;
211 marker.scale.z = 0.05;
212 marker.color.r = 0.0f;
213 marker.color.g = 0.0f;
214 marker.color.b = 1.0f;
215 marker.color.a = 1.3f;
216 return marker;
217}
